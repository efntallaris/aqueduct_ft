\begin{algorithm}[H]
\footnotesize
\caption{Donor Leader Recovery}
\begin{algorithmic}[1]
\Statex \textbf{upon} LEADER\_ELECTION\_COMPLETE \textbf{do}
  \If{self.group\_type $\neq$ DONOR or self $\neq$ LEADER}
    \State \textbf{return} \Comment{Only new donor leader recovers}
  \EndIf
  
  \State committed\_groups $\leftarrow$ RAFT\_QUERY\_COMMITTED\_$G_i$()
  \State active\_groups $\leftarrow$ GET\_ACTIVE\_MIGRATION\_GROUPS()
  
  \For{each $G_i$ in active\_groups}
    \If{$G_i$\_DONE $\in$ committed\_groups}
      \State MARK\_GROUP\_COMPLETE($G_i$)
      \State \textbf{continue}
    \EndIf
    
    \Statex \textit{// Ask recipient about migration progress}
    \State SEND\_CONTROL(recipient\_leader, QUERY\_$G_i$\_STATE($G_i$))
    \State state $\leftarrow$ WAIT\_REPLY(recipient\_leader)
    
    \If{state $=$ MAJORITY\_REPLICATED}
      \Statex \textit{// Recipient has majority $\rightarrow$ safe to commit locally}
      \State RAFT\_APPEND\_LOG($G_i$\_DONE)
      \State RAFT\_COMMIT()
      \State MARK\_GROUP\_COMPLETE($G_i$)
    \Else
      \Statex \textit{// No majority yet $\rightarrow$ retry migration}
      \State RESTART\_MIGRATION($G_i$)
    \EndIf
  \EndFor
\Statex \textbf{end upon}
\end{algorithmic}
\normalsize
\end{algorithm}
