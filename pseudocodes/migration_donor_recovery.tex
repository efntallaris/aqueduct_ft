\begin{algorithm}[H]
\footnotesize
\caption{Donor Leader Recovery}
\begin{algorithmic}[1]
\Statex \textbf{upon} LEADER\_ELECTION\_COMPLETE \textbf{do}
  \If{NOT (self.group\_type is DONOR and self is LEADER)}
    \State \textbf{return} \Comment{Only new donor leader recovers}
  \EndIf

  \Statex \textit{// Check if there is any active migration txn}
  \If{$\neg$ HAS\_ACTIVE\_MIGRATION\_TXN()}
    \State \textbf{return}
  \EndIf

  \Statex \textit{// Load migration progress from Raft}
  \State started\_groups $\leftarrow$ RAFT\_QUERY\_Gi\_WITH\_STATE(STARTED)
  \State done\_groups $\leftarrow$ RAFT\_QUERY\_Gi\_WITH\_STATE(DONE)

  \For{each $G_i$ in started\_groups}

    \If{$G_i \in$ done\_groups}
      \Statex \textit{// Already safely completed before failure}
      \State MARK\_GROUP\_COMPLETE($G_i$)
      \State \textbf{continue}
    \EndIf

    \Statex \textit{// Query recipient leader for current $G_i$ state}
    \State SEND\_RPC(recipient\_leader, QUERY\_Gi\_STATE($G_i$))
    \State state $\leftarrow$ WAIT\_REPLY(recipient\_leader)

    \If{state $=$ Gi\_Done}
      \Statex \textit{// Recipient has quorum durability $\rightarrow$ finalize locally}
      \State RAFT\_APPEND\_LOG(Gi\_DONE)
      \State RAFT\_COMMIT()

    \ElsIf{state $=$ TRANSFERRED}
      \Statex \textit{// RDMA completed, replication still ongoing}
      % \State CONTINUE\_WAITING($G_i$)
      \State WAIT\_FOR\_Gi\_DONE($G_i$)

    \ElsIf{state $=$ PREPARED}
      \Statex \textit{// Buffers exist but no data $\rightarrow$ re-issue RDMA WRITEs}
      \State START\_RDMA\_TRANSFER($G_i$)
      \State WAIT\_FOR\_COMPLETION($G_i$)

    \Else
      \Statex \textit{// Unknown or transient state $\rightarrow$ skip and revisit later}
      \State DEFER\_Gi($G_i$)
    \EndIf

  \EndFor

  \Statex \textit{// Global txn completion check}
  \If{ALL\_GROUPS\_DONE()}
    \State RAFT\_APPEND\_LOG(TXN\_MIGRATION\_DONE)
    \State RAFT\_COMMIT()
  \EndIf

  \Statex \textbf{end upon}
\end{algorithmic}
\normalsize
\end{algorithm}
