\begin{algorithm}[H]
\footnotesize
\caption{Recipient Leader Recovery}
\begin{algorithmic}[1]
\Statex \textbf{upon} LEADER\_ELECTION\_COMPLETE \textbf{do}
  \If{self.group\_type $\neq \texttt{RECIPIENT}$ \textbf{or} self $\neq \texttt{LEADER}$}
    \State \textbf{return}
  \EndIf

  \Statex \textit{// Groups for which this recipient has metadata or data but INDEX\_UPDATE not yet committed}
  \State active\_groups $\leftarrow$ QUERY\_Gi\_WITH\_STATE(PREPARED) OR WITH\_STATE(TRANSFERRED)

  \For{\textbf{each} $G_i$ \textbf{in} active\_groups}
    \State WAIT\_UNTIL\_MAJORITY\_OF\_RDMA\_RECEIVED[Ri][Gi] = true
    \If{RDMA\_RECEIVED[self][Gi] == false}
      \State ASK\_THE\_DATA\_FROM\_DONOR
      \State WAIT\_UNTIL\_RDMA\_DONE\_OF\_Gi
    \EndIf

    \State RAFT\_APPEND(\texttt{INDEX\_UPDATE}($G_i$))
    \State RAFT\_COMMIT()
    \State APPLY\_UPDATE\_INDEX($G_i$)

  \EndFor
  \Statex \textbf{end upon}

  \Statex

\end{algorithmic}
\normalsize
\end{algorithm}
