\begin{algorithm}[H]
\footnotesize
\caption{Recipient Leader Handling of Migrated Data}
\begin{algorithmic}[1]
\Statex \textbf{upon} RECEIVE PREPARE($G_i$, metadata) \textbf{from} donor\_leader \textbf{do}
  \State buffers $\leftarrow$ ALLOCATE\_BUFFERS($G_i$, metadata)
  \State REGISTER\_RDMA(buffers)
  \State STORE\_METADATA($G_i$, metadata, buffers)
\State SEND\_RPC(donor\_leader, PREPARE\_OK($G_i$))
\Statex \textbf{end upon}

\Statex

\Statex \textbf{upon} RECEIVE RDMA\_DONE($G_i$) \textbf{from} donor\_leader \textbf{do}
  \State $\texttt{RDMA\_RECEIVED}[\texttt{self}][G_i] \leftarrow \texttt{true}$
  \Statex \textit{// Build replication chain}
  \State alive\_replicas $\leftarrow [\,]$
  \For{each replica $r$ in REPLICA\_SET($G_i$)}
    \If{CHECK\_ALIVE($r$)}
      \State alive\_replicas.append($r$)
    \EndIf
  \EndFor

  \State chain $\leftarrow$ BUILD\_CHAIN(alive\_replicas)

  \State REPLICATE\_TO($r$, $G_i$)

  
  \State WAIT UNTIL MAJORITY OF ($\texttt{RDMA\_RECEIVED}[R_i][G_i] = \texttt{true}$)
  \State RAFT\_APPEND\_LOG($G_i$\_INDEX\_UPDATE)
  \State RAFT\_COMMIT()
  \State EXECUTE\_UPDATE\_INDEX($G_i$)

  \Statex \textit{// Notify donor leader}
\State SEND\_RPC(donor\_leader, G\_i\_Done($G_i$))
\Statex \textbf{end upon}

\Statex

\Statex \textbf{upon} APPEND\_ENTRIES(entry) from R\_i \textbf{do}

  \If{entry.metadata $= \texttt{RDMA\_DONE}(\texttt{txn\_id}, G_i)$}
      \State{$\texttt{RDMA\_RECEIVED}[R_i][G_i] = \texttt{true}$}
  \EndIf

  % \If{entry.type $= \texttt{UPDATE\_INDEX}(\texttt{txn\_id}, G_i)$}
  %   \If{$\texttt{RDMA\_RECEIVED}[self][G_i] = \texttt{false}$}
  %     \State REJECT\_APPEND()
  %     \State \textbf{return}
  %   \EndIf
  % \EndIf
  \State APPEND\_TO\_LOG(entry)
  \State REPLY\_SUCCESS()
\Statex \textbf{end upon}
\end{algorithmic}
\normalsize
\end{algorithm}
